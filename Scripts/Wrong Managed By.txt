SELECT REF_NO DOCUMENTNO,To_Date(REFDATE,'DD/MM/RRRR') DOCUMENTDATE,AMOUNT,
IPOL.POLICYID FROM IM_NGI_JVPOSTING JV
 JOIN IM_INDIVIDUALPOLICY IPOL ON IPOL.INDIVIDUALPOLICYCODE = JV.POLICYCODE WHERE REF_CODE IN
(SELECT POLICYFINANCEPOSTINGCODE FROM IM_POLICYFINANCEPOSTING WHERE INDIVIDUALPOLICYCODE IN
(SELECT INDIVIDUALPOLICYCODE FROM IM_INDIVIDUALPOLICY WHERE OWNERCODE = 11))
AND JVPREMIUMTYPE <> 0   AND JV.AGENTCODE IS NULL
