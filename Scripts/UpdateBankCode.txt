BEGIN
FOR HDR IN (SELECT * FROM UPDATEBANK) LOOP
UPDATE GENCONSTANT SET SHORTNAME = HDR.SHORTNAME WHERE CATEGORY='SP_BANKS' AND CONSTANTVALUE=HDR.CONSTANTVALUE;
COMMIT;
UPDATE FND_LOOKUP_VALUES SET ARABIC_VALUE = HDR.SHORTNAME WHERE LOOKUP_TYPE='SP_BANKS' AND To_Char(LOOKUP_CODE)=To_Char(HDR.CONSTANTVALUE);
COMMIT;
END LOOP;
END;
