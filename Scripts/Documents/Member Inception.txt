BEGIN
FOR HDR IN(SELECT MEMBER_CODE,A.INCEPTION_DATE,EFFECTIVE_START_DATE FROM IM_INDIVIDUALPOLICY IP,IM_IP_MEMBERS A,IM_MEMBERS M WHERE A.MEMBERCODE = M.MEMBER_CODE
AND A.INCEPTION_DATE <> EFFECTIVE_START_DATE  AND IP.INDIVIDUALPOLICYCODE = A.INDIVIDUALPOLICYCODE AND IP.PORTALTYPE='PR')LOOP
UPDATE IM_MEMBERS SET EFFECTIVE_START_DATE = HDR.INCEPTION_DATE WHERE MEMBER_CODE = HDR.MEMBER_CODE;
COMMIT;
UPDATE IM_MEMBERPOLICY SET MEMBERSTARTDATE = HDR.INCEPTION_DATE WHERE MEMBERCODE = HDR.MEMBER_CODE;
COMMIT;
END LOOP;
END;




BEGIN

AND A.INCEPTION_DATE <> EFFECTIVE_START_DATE  AND IP.POLICYCODE = A.POLICYCODE AND IP.PORTALTYPE='PR')LOOP
UPDATE IM_MEMBERS SET EFFECTIVE_START_DATE = HDR.INCEPTION_DATE WHERE MEMBER_CODE = HDR.MEMBER_CODE;
COMMIT;
UPDATE IM_MEMBERPOLICY SET MEMBERSTARTDATE = HDR.INCEPTION_DATE WHERE MEMBERCODE = HDR.MEMBER_CODE;
COMMIT;
END LOOP;
END;
