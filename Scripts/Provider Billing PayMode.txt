SELECT PRO.PROVIDERID,PRO.PROVIDERNAME,PRD.STARTDATE,PRD.ENDDATE,GEN.CONSTANTNAME,Decode(PRD.PAYMODE,1,'OB',2,'TR',3,'TT',NULL) FROM IM_PROVIDERS PRO
LEFT JOIN IM_PROVIDER_BILLING_DETAIL PRD ON PRD.PROVIDERCODE = PRO.PROVIDERCODE
LEFT JOIN GENCONSTANT GEN ON GEN.CATEGORY = 'SP_BANKS' AND GEN.CONSTANTVALUE = PRD.BANKCODE AND GEN.LANGUAGECODE = 'en-US'
WHERE Upper(PRO.PROVIDERNAME) NOT LIKE '%DUMMY%'    AND SubStr(PRO.PROVIDERID,0,1) <> 'G'
ORDER BY PROVIDERID
