SELECT IMP.AMEMBERNAME,IMP.ACARDID,PRO.PROVIDERID,PRO.PROVIDERNAME,IMC.NAME,
IVD.CODE,IVD.SHORTDESC,ECA.QUANTITY
FROM IM_CLAIM_PROCESS_HEADER HDR,ECLAIM_ACTIVITIES  ECA,IM_MEMBERPOLICY IMP,
IM_PROVIDERS PRO,IM_CLINICIANS IMC,IM_VERSION_DETALIS IVD
WHERE HDR.MAP_VALUE = ECA.SYS_CLAIM_ID   AND HDR.PROVIDER_CODE = PRO.PROVIDERCODE AND PRO.PROVIDERTYPE = 2
AND REQUEST_AMOUNT > 1000 AND To_Date(TRANSACTIONDATE,'DD/MM/RRRR') BETWEEN
To_Date('01/03/2016','DD/MM/RRRR') AND To_Date('14/05/2016','DD/MM/RRRR') 
AND Nvl(ECA.CLINICIAN,ECA.ORDERINGCLINICIAN) = IMC.LICENSEID AND 
HDR.MEMBERPOLICYCODE = IMP.MEMBERPOLICYCODE AND IVD.CODE = ECA.ACTIVITY_CODE;
