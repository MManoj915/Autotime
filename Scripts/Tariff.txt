SELECT * FROM IM_PROVIDERS WHERE PROVIDERID='A057'


SELECT IPTD.* FROM IM_PROVIDER_TARIFF IPT,IM_PROVIDER_TARIFF_DETAILS IPTD
WHERE PROVIDERCODE=1000000000000000053 AND IPT.TARIFFCODE = IPTD.TARIFFCODE
ORDER BY  IPTD.EFFECTIVETO DESC
 
AND SYSDATE BETWEEN IPTD.EFFECTIVEFROM AND IPTD.EFFECTIVETO


UPDATE IM_PROVIDER_TARIFF_DETAILS SET EFFECTIVETO = To_Date('31/12/2025','DD/MM/RRRR')
WHERE EFFECTIVETO = To_Date('31/12/2015','DD/MM/RRRR')    AND TARIFFCODE IN
(SELECT TARIFFCODE FROM IM_PROVIDER_TARIFF WHERE PROVIDERCODE =  1000000000000000053)
