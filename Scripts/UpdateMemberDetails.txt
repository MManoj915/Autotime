DECLARE L_MEMBERCODE NUMBER;
BEGIN
FOR HDR IN(SELECT * FROM IM_MEM_UPDATE) LOOP
SELECT Max(MEMBERCODE) INTO L_MEMBERCODE FROM IM_MEMBERPOLICY WHERE Upper(AMEMBERNAME) = Upper(HDR.MEMBERNAME);
UPDATE IM_MEMBERPOLICY SET A_EMIRATES_ID=HDR.EMIRATESID,UIDNUMBER=HDR.UIDNUMBER,A_EMAILID=HDR.EMAILID
WHERE Upper(AMEMBERNAME) = Upper(HDR.MEMBERNAME) AND APOLICYID = '8218-1';
COMMIT;
UPDATE IM_MEMBERS SET EMIRATES_ID=HDR.EMIRATESID,UIDNUMBER=HDR.UIDNUMBER,EMAILID=HDR.EMAILID
WHERE MEMBER_CODE = L_MEMBERCODE AND APOLICYID = '8218-1';
COMMIT;  
END LOOP;
END;
