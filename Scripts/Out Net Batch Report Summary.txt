
SELECT ICB.FROMID BATCHNO,ICB.RECEIVEDON RECEIPTDATE,ICMB.RECEIPTNUMBER,
ICMB.ACR_FORM_NO FORMNO,ICP.CARDNO,MEM.AMEMBERNAME INSURERNAME,
NVL(POL.POLICYID,IPOL.POLICYID) POLICYID,GEN.SHORTNAME CURRENCYNAME,
ICMB.REQUIREDAMOUNT INVOICEAMOUNT,ICMB.PROCESSEDAMOUNT,ICMB.DENIALAMOUNT FROM IM_CLAIMMEMBERBATCH ICMB
JOIN IM_CLAIMBATCH ICB ON ICB.CLAIMBATCHCODE = ICMB.CLAIMBATCHCODE
JOIN IM_MEMBERPOLICY MEM ON MEM.MEMBERCODE = ICMB.MEMBERCODE AND ICMB.POLICYCODE = MEM.POLICYCODE AND ICMB.CATEGORYCODE = MEM.CATEGORYCODE
JOIN IM_CORDPRINT ICP ON ICP.MEMBERCODE = ICMB.MEMBERCODE
LEFT JOIN IM_POLICY POL ON POL.POLICYCODE = MEM.POLICYCODE AND MEM.TYPEE = 1
LEFT JOIN IM_INDIVIDUALPOLICY IPOL ON IPOL.INDIVIDUALPOLICYCODE = MEM.POLICYCODE AND MEM.TYPEE = 2
JOIN GENCURRENCY GEN ON GEN.CURRENCYCODE = ICMB.CURRENCY
WHERE  ICMB.CLAIMBATCHCODE = 100000000000049411
