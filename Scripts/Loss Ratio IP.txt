SELECT TOTALIPCLAIMPAID,0 OUTSTANDINGCLAIMSAMOUNT,TOTALIPPREMIUM,ROUND((TOTALIPPREMIUM/365) * NVL(EARNEDDAYS,0),2) EARNEDIPPREMIUM
FROM (SELECT (SELECT  SUM(Nvl(HDR.APPROVED_AMOUNT,0))   FROM IM_CLAIM_PROCESS_HEADER HDR
WHERE HDR.POLICYCODE=POL.POLICYCODE AND Nvl(HDR.TREATMENT_TYPE,0) = 0) TOTALIPCLAIMPAID,
GET_POLICYIPPREMIUM(POL.POLICYCODE) TOTALIPPREMIUM,ROUND((SYSDATE-POL.STARTDATE)) EARNEDDAYS
FROM IM_POLICY POL WHERE POL.POLICYID = '0739-1')
