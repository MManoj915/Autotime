SELECT M.ACARDID,M.AMEMBERNAME,M.APOLICYID,M.APOLICYNAME,
H.ACR_FORM_NO,H.INVOICENUMBER,H.TRANSACTIONDATE,H.ENCOUNTER_START_DATE,
H.REQUEST_AMOUNT CLAIMAMOUNT,H.DENAILVALUE,H.APPROVED_AMOUNT PAIDAMOUNT,
G.CURRENCYNAME,G.SHORTNAME,H.PAYMENTREFNO,H.PAYMENTREFDATE
FROM IM_CLAIM_PROCESS_HEADER H,IM_MEMBERPOLICY M,GENCURRENCY G
WHERE REQUEST_TYPE = 3 AND H.CLAIM_STATUS = 2 AND M.MEMBERPOLICYCODE = H.MEMBERPOLICYCODE
AND H.CURRENCYCODE = G.CURRENCYCODE
AND TO_DATE(H.CREATEDON,'DD/MM/RRRR') BETWEEN TO_DATE('01/01/2017','DD/MM/RRRR')
AND TO_DATE('31/12/2017','DD/MM/RRRR')
