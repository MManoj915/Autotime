SELECT ROWNUM AS AMEMBERBATCHCODE,ICB.FROMID BATCHNO,ICMB.ACR_FORM_NO FORMNO,
ICMB.RECEIPTNUMBER,ICB.RECEIVEDON RECEIPTDATE,NVL(POL.POLICYID,IPOL.POLICYID) POLICYID,
ICP.CARDNO,MEM.AMEMBERNAME INSURERNAME,(ICMB.REQUIREDAMOUNT || ' ' || CUR.SHORTNAME) INVOICEAMOUNT,ICMB.REMARKS REMARKS FROM IM_CLAIMMEMBERBATCH ICMB
JOIN IM_CLAIMBATCH ICB ON ICB.CLAIMBATCHCODE = ICMB.CLAIMBATCHCODE
JOIN IM_MEMBERPOLICY MEM ON MEM.MEMBERCODE = ICMB.MEMBERCODE AND ICMB.POLICYCODE = MEM.POLICYCODE AND ICMB.CATEGORYCODE = MEM.CATEGORYCODE
JOIN IM_CORDPRINT ICP ON ICP.MEMBERCODE = ICMB.MEMBERCODE AND ICP.POLICYCODE = MEM.POLICYCODE AND ICP.CATEGORYCODE = MEM.CATEGORYCODE
LEFT JOIN IM_POLICY POL ON POL.POLICYCODE = MEM.POLICYCODE AND MEM.TYPEE = 1
LEFT JOIN IM_INDIVIDUALPOLICY IPOL ON IPOL.INDIVIDUALPOLICYCODE = MEM.POLICYCODE AND MEM.TYPEE = 2
JOIN GENCURRENCY CUR ON CUR.CURRENCYCODE = ICMB.CURRENCY
WHERE  ICMB.CLAIMBATCHCODE = 100000000000049411
