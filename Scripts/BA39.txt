SELECT  MPOL.ACARDID CARDNO,MPOL.AMEMBERNAME MEMBERNAME,
(SELECT LISTAGG (IMVD.SHORTDESC, ',')WITHIN GROUP (ORDER BY IMVD.SHORTDESC)
FROM ECLAIM_DIAGNOSIS DTL 
JOIN IM_VERSION_DETALIS IMVD ON IMVD.CODE = DTL.DIAGNOSIS_CODE
WHERE  DTL.SYS_CLAIM_ID = EC.SYS_CLAIM_ID
GROUP BY DTL.SYS_CLAIM_ID)ICDCODES,
(SELECT IMVD.NAME
FROM ECLAIM_ACTIVITIES DTL 
JOIN IM_CLINICIANS IMVD ON IMVD.LICENSEID = Nvl(DTL.CLINICIAN,DTL.ORDERINGCLINICIAN)
WHERE  DTL.SYS_ACTIVITY_ID = ECA.SYS_ACTIVITY_ID AND ROWNUM < 2)CLINICIANNAME,
ECA.ACTIVITY_CODE DRUGCODE,
(SELECT DISTINCT IMVD.SHORTDESC
FROM ECLAIM_ACTIVITIES DTL 
JOIN IM_VERSION_DETALIS IMVD ON IMVD.CODE = DTL.ACTIVITY_CODE
WHERE  DTL.SYS_ACTIVITY_ID = ECA.SYS_ACTIVITY_ID  AND ROWNUM < 2) DRUGNAME,
ECA.ACTIVITY_NET,ECA.QUANTITY
FROM ECLAIMS EC,IM_MEMBERPOLICY MPOL,ECLAIM_ACTIVITIES ECA
WHERE UPPER(TRIM(REPLACE(MPOL.ACARDID,' ')))=UPPER(TRIM(REPLACE(EC.MEMBER_ID,' ',''))) 
AND EC.SYS_CLAIM_ID = ECA.SYS_CLAIM_ID
AND EC.SENDER_ID = 'PF1426' AND 
EC.TRANSACTION_DATE BETWEEN To_Date('01/11/2015','DD/MM/RRRR') AND To_Date('23/05/2016','DD/MM/RRRR')
AND DISPOSITION_FLAG <> '0' AND SYS_CLAIM_STATUS <> 4
