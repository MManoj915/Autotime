Group Master :

SELECT IG.*,GENCO.CONSTANTNAME AS BILLINGCURRENCYNAME,GEN.CONSTANTNAME AS STATUSNAME, PARGRO.GROUP_NAME AS PARENTGROUPNAME,
GEN1.CONSTANTNAME AS PRINTGROUPTYPENAME FROM IM_GROUPS IG                        
LEFT JOIN GENCONSTANT GENCO ON GENCO.CONSTANTVALUE=IG.CURRENCY_CODE  AND GENCO.CATEGORY='BILLINGCURRENCY' 
AND Upper(GENCO.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=IG.STATUS  AND GEN.CATEGORY='PROVIDERSTATUS' AND Upper(GEN.LANGUAGECODE)=Upper('EN-US')
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=IG.PRINTGROUPTYPE  AND GEN1.CATEGORY='PRINTGROUPTYPE' AND 
Upper(GEN1.LANGUAGECODE)=Upper('EN-US')   
LEFT JOIN IM_GROUPS PARGRO ON PARGRO.GROUP_CODE=IG.PARENT_GROUP;

Group Billing : 

SELECT BIL.*,GRO.GROUP_NAME GROUPNAME,
GEN.CONSTANTNAME AS AUTHORIZEDSTATUSNAME,  CON.COUNTRYNAME COUNTRYNAME,PLACE.PLACENAME CITYNAME,STATE.PROVINCENAME STATENAME,
BANK.CONSTANTNAME BANKNAME,GEC.CURRENCYNAME CURRENCYNAME FROM IM_GROUP_BILLING_DETAIL BIL   
LEFT JOIN IM_GROUPS GRO ON GRO.GROUP_CODE=BIL.GROUPCODE   
LEFT JOIN GENCONSTANT BANK ON BANK.CONSTANTVALUE=BIL.BANKCODE AND CATEGORY='SP_BANKS' AND Upper(LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' AND 
Upper(LANGUAGECODE)=Upper('EN-US'))GEN ON GEN.CONSTANTVALUE=BIL.AUTHORIZEDSTATUS   
LEFT JOIN GENCOUNTRY CON ON CON.COUNTRYCODE=BIL.COUNTRYCODE  
LEFT JOIN GENPLACE PLACE ON PLACE.PLACECODE=BIL.CITYCODE  
LEFT JOIN GENPROVINCE STATE ON STATE.PROVINCECODE=BIL.STATECODE  
LEFT JOIN GENCURRENCY GEC ON GEC.CURRENCYCODE=BIL.CURRENCYCODE  ORDER BY BIL.GBDCODE DESC;
