Plan Master :

SELECT PLAN.*,GEN.CONSTANTNAME AS STATUSNAME,GEN1.CONSTANTNAME OWNERTYPENAME,0 AS COPYPLANCODE,'' AS COPYPLANNAME,
'' AS COPYBENEFITBUTTON,'' AS CARDTEXTAREA,'' AS CARDTEXT FROM IM_PLANS PLAN  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE= PLAN.PLAN_STATUS AND GEN.CATEGORY='PROVIDERSTATUS' AND GEN.LANGUAGECODE=UPPER('EN-US') LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=PLAN.OWNERCODE AND GEN1.CATEGORY='NGIQUOTATIONTYPE' AND Upper(GEN1.LANGUAGECODE)=UPPER('EN-US');


Area Details :

SELECT AREA.*,GAREA.GANAME,GAREA.GAID,GEN.CONSTANTNAME AS STATUSNAME FROM IM_PLAN_AREAS AREA 
LEFT JOIN IM_GEOGRAPHIC_AREAS GAREA ON GAREA.GACODE=  AREA.GACODE 
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE= AREA.STATUS AND GEN.CATEGORY='PROVIDERSTATUS' AND Upper(GEN.LANGUAGECODE)=UPPER('EN-US')  


Network Details :

SELECT PNET.*,NET.NETWORKID,NET.NETWORKNAME,GEN1.CONSTANTNAME AS STATUSNAME,GEN.CONSTANTNAME APPLICABLEFORNAME 
FROM  IM_PLAN_NETWORKS PNET 
LEFT JOIN IM_NETWORKS NET ON NET.NETWORKCODE =PNET.NETWORKCODE 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE = PNET.STATUS AND GEN1.CATEGORY='PROVIDERSTATUS' AND GEN1.LANGUAGECODE=UPPER('EN-US')LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE = PNET.APPLICABLEFORCODE AND GEN.CATEGORY='APPLICABLE' AND Upper(GEN.LANGUAGECODE)=UPPER('EN-US') 


Benefit Details :

SELECT PBEN.*,BEN.BENEFIT_ID,BEN.BENEFIT_NAME, BEN.DEFAULTPREMIUM, GEN.CONSTANTNAME TYPENAME,GEN1.CONSTANTNAME STATUSNAME,
(SELECT BEN1.BENEFIT_NAME BENEFITNAME FROM IIM_BENEFIT_CODES_DET SUBDET  
JOIN IM_BENEFIT_CODES BEN1 ON BEN1.BENEFIT_CODE=SUBDET.BENEFIT_CODE 
WHERE SUBDET.SUB_BENEFIT_CODE= PBEN.BENEFIT_CODE AND ROWNUM=1)PARENTNAME FROM IM_PLAN_BENEFITS PBEN   
LEFT JOIN IM_BENEFIT_CODES BEN ON BEN.BENEFIT_CODE=PBEN.BENEFIT_CODE 
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE = PBEN.BENEFIT_TYPE AND GEN.CATEGORY='BENEFITTYPE' AND Upper(GEN.LANGUAGECODE)=UPPER('EN-US')   
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE = PBEN.STATUS AND GEN1.CATEGORY='PROVIDERSTATUS' AND Upper(GEN1.LANGUAGECODE)=UPPER('EN-US');

Emirate Details :

SELECT EMI.*,PRO.PROVINCENAME EMIRATENAME FROM IM_PLANEMIRATES  EMI  
JOIN GENPROVINCE PRO ON PRO.PROVINCECODE=EMI.EMIRATECODE;

Emirate sub Details :

SELECT AGE.*,GEN.CONSTANTNAME GENDERNAME,GEN1.CONSTANTNAME APPLICABLEFORNAME FROM IM_PLANAGELIMITS AGE     
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=AGE.APPLICABLEFOR AND GEN1.CATEGORY='MARITAL_STATUS' AND Upper(GEN1.LANGUAGECODE)=Upper(UPPER('EN-US'))  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=AGE.GENDERCODE AND GEN.CATEGORY='GENDER' AND Upper(GEN.LANGUAGECODE)=UPPER('EN-US')
