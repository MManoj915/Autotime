SELECT AMU.LOGCODE,AMU.SERIALNO,AMU.ADDMEMBERPOLICYCODE,(ADDMEM.BILLINGNO ||' - '|| NVL(POL.POLICYNAME,INDPOL.POLICYNAME)) AS ADDMEMBERREFNO,
AMU.REFDATE,AMU.UPLOADON,GEN.CONSTANTNAME AS UPLOADTYPENAME,
EUT.UPLOADNAME UPLOADTEMPLATENAME,AMU.UPLOADEDRECORDCOUNT,AMU.IMPORTEDRECORDCOUNT  FROM IM_ADDMEMBER_UPLOAD AMU    
LEFT JOIN IM_ADDMEMBERPOLICY  ADDMEM ON ADDMEM.ADDMEMBERPOLICYCODE=AMU.ADDMEMBERPOLICYCODE    
LEFT JOIN IM_POLICY  POL ON POL.POLICYCODE=ADDMEM.POLICYCODE  AND ADDMEM.POLICYTYPE=1  
LEFT JOIN IM_INDIVIDUALPOLICY  INDPOL ON INDPOL.INDIVIDUALPOLICYCODE=ADDMEM.INDIVIDUALPOLICYCODE  AND ADDMEM.POLICYTYPE=2  
LEFT JOIN IM_EXCEL_UPLOAD_TEMP_HDR EUT ON EUT.UPLOADCODE=AMU.UPLOADTEMPLATECODE    
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=AMU.UPLOADTYPE AND GEN.CATEGORY='UPLOADTYPE' AND LANGUAGECODE='en-US';
