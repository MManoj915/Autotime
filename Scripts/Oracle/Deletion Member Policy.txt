Deletion Member Policy Master :

SELECT DC.*, GEN1.CONSTANTNAME AUTHORIZEDSTATUSNAME,GEN2.CONSTANTNAME TYPENAME,GEN3.CONSTANTNAME POLICYTYPENAME,
IG.GROUP_NAME GROUPNAME,PO.POLICYID,PC.POLICYNAME,PC.STARTDATE,  PC.ENDDATE,PC.BASE_PERIMUM,PC.ALTLIMIT,PC.OPLIMIT,PC.IPLIMIT, 
GEN4.CONSTANTNAME ANNUALLIMITTYPE,  GEN5.CONSTANTNAME INPATIENTTYPE,   GEN6.CONSTANTNAME OUTPATIENTTYPE, 
GENCAN.CONSTANTNAME AS CANCELLEDBYNAME,IPO.POLICYNAME INDIVIDUALPOLICYNAME,IPO.POLICYID IPPOLICYID,
GEE.CONSTANTNAME POLICYTYPESNAME,'' UPLOADDELMEMBER,INS.INS_NAME INSURANCENAME,GEN8.CONSTANTNAME CONTRACTTYPENAME FROM 
IM_DEL_CANCEL_MEMBER DC   
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CATEGORY='AUTHORIZEDSTATUS' AND GEN1.LANGUAGECODE='EN-US' AND 
GEN1.CONSTANTVALUE=DC.AUTHORIZEDSTATUS   
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CATEGORY='DELCANCEL' AND GEN2.LANGUAGECODE='EN-US'  AND 
GEN2.CONSTANTVALUE=DC.TYPECODE   
LEFT JOIN (SELECT PO.POLICYCODE POLICYCODE, PO.POLICYID,PO.POLICYNAME,PO.STARTDATE,PO.ENDDATE,PO.BASE_PERIMUM,
PO.ALTLIMIT ALTLIMIT,PO.IPLIMIT IPLIMIT,PO.OPLIMIT OPLIMIT,1 TYPECODE,PO.CATEGORYCODE,PO.GROUPCODE,PO.PERIMUMCODE,
PO.ANNUALLIMITTYPE ANNUALLIMITTYPE,PO.INPATIENT INPATIENT,PO.OUTPATIENT OUTPATIENT FROM IM_POLICY PO 
UNION ALL 
SELECT IPO.INDIVIDUALPOLICYCODE POLICYCODE,IPO.POLICYID, IPO.POLICYNAME,IPO.STARTDATE,IPO.ENDDATE,IPO.BASE_PERIMUM,
IPO.ALTLIMITS ALTLIMIT,IPO.IPTLIMITS IPLIMIT,IPO.OPTLIMITS,2 TYPECODE,IPO.CATEGORYCODE,IPO.GROUPCODE,IPO.PERIMUMCODE,
IPO.ANUALLIMITSTYPECODE ANNUALLIMITTYPE,IPO.INPATENTSLIMITSTYPECODE INPATIENT,IPO.OUTPATENTSLIMITSTYPECODE OUTPATIENT 
FROM IM_INDIVIDUALPOLICY IPO)PC ON PC.POLICYCODE=DECODE(DC.POLICYTYPECODE,1,DC.POLICYCODE,2,DC.INDIVIDUALPOLICYCODE)AND PC.TYPECODE=DC.POLICYTYPECODE    
LEFT JOIN IM_GROUPS IG ON IG.GROUP_CODE=PC.GROUPCODE   
LEFT JOIN GENCONSTANT  GEN3 ON GEN3.CONSTANTVALUE=PC.PERIMUMCODE AND GEN3.CATEGORY='PERIMUMS' AND GEN3.LANGUAGECODE='EN-US'   
LEFT JOIN GENCONSTANT  GENCAN ON GENCAN.CONSTANTVALUE=DC.CANCELLEDBYCODE AND GENCAN.CATEGORY='CANCELLEDBY' AND 
GENCAN.LANGUAGECODE='EN-US'   
LEFT JOIN GENCONSTANT GEN4 ON GEN4.CONSTANTVALUE=PC.ANNUALLIMITTYPE AND GEN4.CATEGORY='CALCMETHOD' AND 
GEN4.LANGUAGECODE='EN-US'   
LEFT JOIN GENCONSTANT GEN5 ON GEN5.CONSTANTVALUE=PC.INPATIENT AND GEN5.CATEGORY='CALCMETHOD' AND GEN5.LANGUAGECODE='EN-US'   
LEFT JOIN GENCONSTANT GEN6 ON GEN6.CONSTANTVALUE=PC.OUTPATIENT AND  GEN6.CATEGORY='CALCMETHOD' AND GEN6.LANGUAGECODE='EN-US' 
LEFT JOIN IM_INDIVIDUALPOLICY IPO ON IPO.INDIVIDUALPOLICYCODE= DC.INDIVIDUALPOLICYCODE   
LEFT JOIN IM_POLICY PO ON PO.POLICYCODE= DC. POLICYCODE  
LEFT JOIN GENCONSTANT GEE ON GEE.CONSTANTVALUE=DC.POLICYTYPECODE AND  GEE.CATEGORY='MEMBERPOLICYTYPE' AND 
GEE.LANGUAGECODE='EN-US' 
LEFT JOIN (SELECT * FROM  IM_INSURER) INS ON INS.INSCODE=PO.INSURANCECO 
LEFT JOIN GENCONSTANT GEN8 ON GEN8.CONSTANTVALUE=PO.CONTRACTTYPE AND GEN8.CATEGORY='CONTRACTINGTYPE' AND 
GEN8.LANGUAGECODE='EN-US' ORDER BY DC.DCMEMBER_CODE DESC;

Member Detail : 

SELECT IQM.*, (IME.FIRST_NAME||' '||IME.LAST_NAME) AS MEMBERNAME,IME.DATE_OF_BIRTH,
FLOOR(MONTHS_BETWEEN(SYSDATE,IME.DATE_OF_BIRTH)/12) AS AGE,GEN.CONSTANTNAME AS GENDERNAME,
GEN1.CONSTANTNAME MEMBERRELATIONNAME,GEN2.CONSTANTNAME MARITALSTATUSNAME,GEN3.CONSTANTNAME AS NATIONALITYNAME,
GC.CATEGORY_NAME CATEGORYNAME,IPC.MEMBERSTARTDATE,IME.PARENT_ID FROM IM_DELCAN_MEMBERSDTL IQM  
LEFT JOIN IM_MEMBERS IME ON IME.MEMBER_CODE= IQM.MEMBERCODE   
LEFT JOIN IM_MEMBERPOLICY IPC ON IPC.MEMBERPOLICYCODE=IME.MEMBERPOLICYCODE  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=IME.GENDER  AND GEN.CATEGORY='GENDER' AND LANGUAGECODE='EN-US'   
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=IME.RELATION AND GEN1.CATEGORY='MEMBERRELATION' AND 
GEN1.LANGUAGECODE='EN-US'  
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=IME.MARITAL_STATUS AND GEN2.CATEGORY='MARITAL_STATUS' 
AND GEN2.LANGUAGECODE='EN-US'  
LEFT JOIN GENCONSTANT GEN3 ON GEN3.CONSTANTVALUE=IME.NATIONALITY AND GEN3.CATEGORY='FND_NATIONALITY' AND 
GEN3.LANGUAGECODE='EN-US'   
LEFT JOIN IM_CATEGORIES GC ON GC.CATEGORY_CODE=IME.CATEGORY_CODE  
ORDER BY NVL(IME.PARENT_ID,IME.MEMBER_CODE),NVL2(IME.PARENT_ID,0,1) DESC ;

Commission Detail :

SELECT PCO.*,GEN.CONSTANTNAME COMISSIONTYPENAME,GEN1.CONSTANTNAME CALCMETHODNAME,COM.COMMISIONNAME,
COMBILL.COMMISSIONNAME COMMISSIONNAMEE FROM IM_DELMEMBERCOMMISIONDET PCO 
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=PCO.COMISSION_TYPES AND GEN.CATEGORY='NGICOMMISSIONTYPE' AND 
GEN.LANGUAGECODE='EN-US' 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=PCO.CALC_METHOD AND GEN1.CATEGORY='FINECALCULATIONTYPE' AND 
GEN1.LANGUAGECODE='EN-US' 
LEFT JOIN IM_GENERAL_COMM_BILLING_DETAIL COMBILL ON COMBILL.GENERALCOMMBILLINGCODE=PCO.GENERALCOMMBILLINGCODE   
LEFT JOIN (SELECT AGE.AGENTCODE COMMISIONTOCODE, AGE.AGENT_ID COMMISIONID, AGE.AGENT_NAME_EN COMMISIONNAME ,1 TYPECODE 
FROM IM_AGENTS AGE UNION ALL 
SELECT BRO.BROKERCODE COMMISIONTOCODE, BRO.BROKER_ID COMMISIONID, BRO.BROKER_NAME_EN COMMISIONNAME,0 TYPECODE 
FROM IM_BROKERS BRO) COM ON  COM.COMMISIONTOCODE=PCO.COMMISSIONTO  AND COM.TYPECODE=PCO.COMISSION_TYPES ;
