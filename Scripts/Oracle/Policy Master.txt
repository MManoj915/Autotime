SELECT POL.*,INS.INS_NAME INSURANCENAME,GEN8.CONSTANTNAME CONTRACTTYPENAME,GEN.CONSTANTNAME ANNUALLIMITTYPENAME,
GEN1.CONSTANTNAME INPATIENTNAME,GEN2.CONSTANTNAME OUTPATIENTNAME,GEN11.CONSTANTNAME REQSOURCENAME,
GEN4.CONSTANTNAME COMISSIONTYPENAME, AGE.AGENT_NAME_EN AGENTNAME, BRO.BROKER_NAME_EN BROKERNAME, 
ORG.ORGANIZATION_NAME BRANCHNAME ,LOC.HRLOCATIONNAME LOCATIONNAME ,AORG.ORGANIZATION_NAME AGENTBRANCHNAME,
ALOC.HRLOCATIONNAME AGENTLOCATIONNAME, 0 POLICYGROUPCOUNT,0 REQUESTGROUPCOUNT,0 QUOTAIONMASTERGROUPCOUNT,  
IG.GROUP_NAME GROUPNAME,(MEM.FIRST_NAME||' '||MEM.LAST_NAME)  MEMBERNAME,MEM.MEMBER_ID MEMBERID,
GEN3.CONSTANTNAME AS AUTHORIZEDSTATUSNAME,GENN2.CONSTANTNAME PAYMENTMODENAME,MEM.ARABIC_FIRST_NAME ALTERNATENAME,
GENN.CONSTANTNAME PERIMUMNAME,TEMP.TEMPLATE_NAME_EN,
UB.UBIPNAME UBIPNAME,GEN10.CONSTANTNAME OWNERTYPENAME,ISU.INS_NAME REINSURANCENAME,ISU.ALTERNATIVE_NAME REALTERNATENAME,   
BILGEN.CONSTANTNAME BILLINGTYPENAME, BRA.ORGANIZATION_NAME BRANCHNAMEE,GENCON.CONSTANTNAME REINSURERTYPENAME,GENCON1.CONSTANTNAME SUBREINSURERTYPENAME 
FROM IM_POLICY POL  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=POL.ANNUALLIMITTYPE AND GEN.CATEGORY='CALCMETHOD' AND Upper(GEN.LANGUAGECODE)=Upper(Upper('EN-US')) 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=POL.INPATIENT AND GEN1.CATEGORY='CALCMETHOD' AND Upper(GEN1.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=POL.OUTPATIENT AND  GEN2.CATEGORY='CALCMETHOD' AND Upper(GEN2.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='SOURCE' AND Upper(LANGUAGECODE)=Upper('EN-US'))GEN11 ON GEN11.CONSTANTVALUE=POL.REQUEST_SOURCE   
LEFT JOIN IM_GROUPS IG ON IG.GROUP_CODE=POL.GROUPCODE 
LEFT JOIN IM_MEMBERS MEM ON MEM.MEMBER_CODE=POL.MEMBERCODE 
LEFT JOIN IM_UBIP UB ON UB.UBIPCODE=POL.UBIPCODE  
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' AND Upper(LANGUAGECODE)=Upper('EN-US'))GEN3 ON GEN3.CONSTANTVALUE=POL.AUTHORIZEDSTATUS 
LEFT JOIN GENCONSTANT  GENN ON GENN.CONSTANTVALUE=POL.PERIMUMCODE AND GENN.CATEGORY='PERIMUMS' AND Upper(GENN.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN4 ON GEN4.CONSTANTVALUE=POL.COMMISIONTYPE AND GEN4.CATEGORY='NGICOMMISSIONTYPE' AND Upper(GEN4.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN IM_AGENTS AGE ON AGE.AGENTCODE=POL.AGENTCODE  
LEFT JOIN IM_BROKERS BRO ON BRO.BROKERCODE=POL.BROKERCODE 
LEFT JOIN HR_ORGANIZATIONS_D ORG ON ORG.ORGANIZATIONS_CODE=BRO.BRANCHCODE   
LEFT JOIN HR_LOCATIONS_D LOC ON LOC.HRLOCATION_CODE=BRO.LOCATION_ID   
LEFT JOIN HR_ORGANIZATIONS_D AORG ON AORG.ORGANIZATIONS_CODE=AGE.BRANCHCODE  
LEFT JOIN HR_LOCATIONS_D ALOC ON ALOC.HRLOCATION_CODE=AGE.LOCATION_ID  
LEFT JOIN GENCONSTANT GEN10 ON GEN10.CONSTANTVALUE=POL.OWNERCODE AND GEN10.CATEGORY='NGIQUOTATIONTYPE' AND UPPER(GEN10.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT  GENN2 ON GENN2.CONSTANTVALUE=POL.PAYMENTMODE AND GENN2.CATEGORY='POLICYPAYMENTMODE' AND Upper(GENN2.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN IM_ELEMENTTEMPLATE TEMP ON TEMP.ELEMENTTEMPLATECODE =POL.ELEMENTTEMPLATECODE 
LEFT JOIN IM_INSURER ISU ON ISU.INSCODE=POL.REINSURANCECODE 
LEFT JOIN (SELECT * FROM  IM_INSURER) INS ON INS.INSCODE=POL.INSURANCECO 
LEFT JOIN GENCONSTANT GEN8 ON GEN8.CONSTANTVALUE=POL.CONTRACTTYPE AND GEN8.CATEGORY='CONTRACTINGTYPE' AND Upper(GEN8.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GENCON ON GENCON.CONSTANTVALUE=POL.REINSURERTYPE AND GENCON.CATEGORY='REINSURERTYPE' AND Upper(GENCON.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GENCON1 ON GENCON1.CONSTANTVALUE=POL.SUBREINSURERTYPE AND GENCON1.CATEGORY='CONTRACTINGTYPE' AND Upper(GENCON1.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT BILGEN ON BILGEN.CONSTANTVALUE=POL.BILLINGTYPE AND BILGEN.CATEGORY='POLICYBILLINGTYPE' AND Upper(BILGEN.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN HR_ORGANIZATIONS_D BRA ON BRA.ORGANIZATIONS_CODE=POL.BRANCHCODE
