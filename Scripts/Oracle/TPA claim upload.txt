TPA Claim Header:

SELECT HDR.TPACLAIMHDRCODE,REFNO,REFDATE,FILENAME,GEN1.CONSTANTNAME AUTHORIZEDSTATUSNAME FROM IM_TPACLAIM_HDR HDR 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=HDR.AUTHORIZEDSTATUS AND GEN1.CATEGORY='AUTHORIZEDSTATUS'
AND GEN1.LANGUAGECODE='en-US' ORDER BY TPACLAIMHDRCODE DESC 


TPA Claim Detail :

SELECT CPH.CLAIM_CODE,CPH.TPACLAIMHDRCODE,
CPH.CPNO CLAIMNUMBER,CPH.ACR_FORM_NO FORMNUMBER,CPH.INVOICENUMBER,
CPH.POLICYID,CPH.POLICYNAME,CPH.POLICYTYPENAME SOURCE,
CPH.TRANSACTIONDATE,CPH.REQUEST_AMOUNT INVOICEAMOUNT,CPH.APPROVED_AMOUNT,
CPH.DENAILVALUE,CPH.STATUSNAME,CPH.NEXTCAREDENIEDFORNAME RECOVERYFROM,
CPH.AUTHORIZEDSTATUSNAME FROM IM_NEXTCARECLAIMS_VW CPH WHERE CPH.TPACLAIMHDRCODE=100000000000001041
