CLAIM TYPE :(Request_TYPE COLUMN IN IM_Claim_Process_Header TABLE)
SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY = 'CLAIMPROSOURCETYPE'
AND Upper(LANGUAGECODE) = 'EN-US';

SOURCE PROVIDER :(SOURCEPROVIDER COLUMN IN IM_Claim_Process_Header TABLE)
SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY = 'PRIORSOURCEPROVIDER'
AND Upper(LANGUAGECODE) = 'EN-US';

(IF Source Provider Value IS 3 THEN need TO CHECK CEEDStatus = 1 ELSE need TO CHECK SOURCEPROVIDER COLUMN);

CLAIM STATUS :(CLAIM_STATUS COLUMN IN IM_Claim_Process_Header TABLE)
SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY = 'CLAIMREQSTATUS'
AND Upper(LANGUAGECODE) = 'EN-US';  

APPROVAL STATUS :(AUTHORIZEDSTATUS COLUMN IN IM_Claim_Process_Header TABLE)
SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY = 'AUTHORIZEDSTATUS'
AND Upper(LANGUAGECODE) = 'EN-US';
