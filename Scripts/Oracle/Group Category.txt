SELECT IGC.*,IG.GROUP_NAME, IC.CATEGORY_NAME,GEN.CONSTANTNAME AS STATUSNAME FROM IM_GROUP_CATEGORIES IGC  
LEFT JOIN IM_GROUPS IG ON IG.GROUP_CODE=IGC.GROUP_CODE LEFT 
JOIN IM_CATEGORIES IC ON IC.CATEGORY_CODE= IGC.CATEGORY_CODE 
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=IG.STATUS  AND GEN.CATEGORY='PROVIDERSTATUS' AND Upper(GEN.LANGUAGECODE)=Upper('EN-US')
