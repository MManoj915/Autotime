,'' UIDNUMBER, '' ISPREEXISTING

Individual Policy Member :

select iqm.*,ime.MEMBER_ID  MemberIDName,GEN.CONSTANTNAME as GenderName,GEN1.CONSTANTNAME MemberRelationName,GEN2.CONSTANTNAME MaritalStatusName, 
 GEN3.CONSTANTNAME as NationalityName,floor(months_between(sysdate,iqm.DATE_OF_BIRTH)/12) as Age,GC.CATEGORY_NAME CategoryName , (select count(*) from IM_IP_PREMIUM det where det.IP_MEM_CODE=iqm.IP_MEM_CODE and det.PREMIUM_TYPES=0) as IsBasePremiumAvail,ParGro.GROUP_NAME as ParentGroupName 
 ,(select sum(PREMIUM_VALUE) from IM_IP_PREMIUM where IP_MEM_CODE=IQM.IP_MEM_CODE) premiums  from IM_IP_MEMBERS iqm  
 left join genconstant gen on GEN.CONSTANTVALUE=iqm.MEMBER_GENDER  and GEN.CATEGORY='Gender' and languagecode='  Helper.LanguageCode  '  
 LEFT join genconstant gen1 on GEN1.CONSTANTVALUE=iqm.MEMBER_RELATION and GEN1.CATEGORY='MEMBERRELATION' and gen1.languagecode='  Helper.LanguageCode  ' 
 left join genconstant gen2 on GEN2.CONSTANTVALUE=iqm.MARITAL_STATUS and GEN2.CATEGORY='MARITAL_STATUS' and GEN2.LANGUAGECODE='  Helper.LanguageCode  ' 
 left join genconstant gen3 on GEN3.CONSTANTVALUE=iqm.NATIONALITY and GEN3.CATEGORY='FND_NATIONALITY' and GEN3.languagecode='  Helper.LanguageCode  '  
 left join IM_MEMBERS ime on iME.MEMBER_CODE= iqm.MemberCode   
 left join im_categories gc on GC.CATEGORY_CODE=iqm.MEMBER_CATEGORY   
 left join (select constantValue,ConstantName from genconstant where category='Gender' and languagecode='en-US')gen4 on gen4.constantValue=iqm.MEMBER_GENDER  
 left join IM_GROUPS ParGro on ParGro.Group_Code=iqm.ParentGroupCode  
  order by nvl(iqm.PARENTID,iqm.MEMBERID),nvl2(iqm.PARENTID,0,1) desc
