SELECT  IMP.NEXTCAREPRODHDRCODE,IMP.SERIALNO,IMP.REINSADDMEMBERPOLICYCODE,
POL.POLICYID,POL.POLICYNAME,POL.STARTDATE,POL.ENDDATE,GROP.GROUP_NAME GROUPNAME,  
GEN3P.CONSTANTNAME MANAGEDBY,GEN.CONSTANTNAME AUTHORIZEDSTATUSNAME FROM IM_REINS_ADDMEMBERPOLICY IMP   
JOIN ADMCOMPANY ADMCOM ON ADMCOM.COMPANYCODE=IMP.COMPANYCODE AND ADMCOM.COMPANYCODE= 1 
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' AND LANGUAGECODE='en-US')GEN 
ON GEN.CONSTANTVALUE=IMP.AUTHORIZEDSTATUS 
LEFT JOIN IM_POLICY POL ON POL.POLICYCODE = IMP.POLICYCODE     
LEFT JOIN IM_GROUPS GROP ON GROP.GROUP_CODE =POL.GROUPCODE         
LEFT JOIN GENCONSTANT GEN3P ON GEN3P.CONSTANTVALUE=IMP.SOURCETYPE AND GEN3P.CATEGORY='REINSURERBATCHTYPE' AND 
GEN3P.LANGUAGECODE='en-US' WHERE IMP.SOURCETYPE = 2 AND IMP.NEXTCAREPRODHDRCODE=100000000000002690 ;

SELECT IQM.REINSADDMEMBERPOLICYCODE,IQM.REINSADDMEMBERDETCODE,IQM.SERIALNO,
IQM.FMCCARDNO CARDNO,IQM.MEMBERID,IQM.MEMBERNAME,IQM.DATE_OF_BIRTH,FLOOR(MONTHS_BETWEEN(SYSDATE,IQM.DATE_OF_BIRTH)/12) AS AGE,
GEN.CONSTANTNAME AS GENDERNAME,GEN1.CONSTANTNAME RELATION,GEN2.CONSTANTNAME MARITALSTATUSNAME,
PARGRO.GROUP_NAME AS SUBGROUP,GC.CATEGORY_NAME CATEGORYNAME,
GEN3.CONSTANTNAME AS NATIONALITYNAME,IQM.EFFECTIVEDATE FROM IM_REINSADDMEMBERDET IQM    
LEFT JOIN IM_MEMBERS IME ON IME.MEMBER_CODE= IQM.MEMBERCODE     
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=IQM.GENDERCODE  AND GEN.CATEGORY='GENDER' AND LANGUAGECODE='en-US'   
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=IQM.MEMBERRELATIONCODE AND GEN1.CATEGORY='MEMBERRELATION' AND 
GEN1.LANGUAGECODE='en-US'   
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=IQM.MARITALSTATUSCODE AND GEN2.CATEGORY='MARITAL_STATUS' AND 
GEN2.LANGUAGECODE='en-US'   
LEFT JOIN GENCONSTANT GEN3 ON GEN3.CONSTANTVALUE=IQM.NATIONALITYCODE AND GEN3.CATEGORY='FND_NATIONALITY' AND 
GEN3.LANGUAGECODE='en-US'  
LEFT JOIN IM_CATEGORIES GC ON GC.CATEGORY_CODE=IQM.CATEGORYCODE 
LEFT JOIN IM_GROUPS PARGRO ON PARGRO.GROUP_CODE=IQM.PARENTGROUPCODE 
WHERE IQM.REINSADDMEMBERPOLICYCODE=100000000000012426;
