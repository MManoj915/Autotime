Agent :

SELECT AGE.*,LOC.HRLOCATIONNAME LOCATIONNAME,CON.COUNTRYNAME,STAT.PROVINCENAME AS STATENAME,PLA.PLACENAME ,
GEN.CONSTANTNAME STATUSNAME,ORG.ORGANIZATION_NAME BRANCHNAME 
FROM IM_AGENTS AGE  LEFT JOIN HR_LOCATIONS_D LOC ON LOC.HRLOCATION_CODE=AGE.LOCATION_ID  
LEFT JOIN GENCOUNTRY CON ON CON.COUNTRYCODE=AGE.AGENT_COUNTRY_ID  
LEFT JOIN GENPROVINCE STAT ON STAT.PROVINCECODE=AGE.AGENT_STATE_ID   
LEFT JOIN GENPLACE PLA ON PLA.PLACECODE=AGE.AGENT_CITY_ID   
LEFT JOIN HR_ORGANIZATIONS_D ORG ON ORG.ORGANIZATIONS_CODE=AGE.BRANCHCODE  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=AGE.AGENT_STATUS  AND GEN.CATEGORY='AGENTSSTATUS' AND Upper(GEN.LANGUAGECODE)=Upper('EN-US');

Agent Billing Detail :

SELECT BIL.*,AGE.AGENT_ID AGENTID,AGE.AGENT_NAME_EN AGENTNAME, AGE.AGENT_NAME_AR ALTERNATEAGETNAME,
GEN.CONSTANTNAME AS AUTHORIZEDSTATUSNAME,  CON.COUNTRYNAME COUNTRYNAME,PLACE.PLACENAME CITYNAME,STATE.PROVINCENAME STATENAME,
BANK.CONSTANTNAME BANKNAME,GEC.CURRENCYNAME CURRENCYNAME FROM IM_AGENTS_BILLING_DETAIL BIL   
LEFT JOIN IM_AGENTS AGE ON AGE.AGENTCODE=BIL.AGENTCODE  
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT 
WHERE CATEGORY='AUTHORIZEDSTATUS' AND Upper(LANGUAGECODE)=Upper('EN-US'))GEN ON GEN.CONSTANTVALUE=BIL.AUTHORIZEDSTATUS   
LEFT JOIN GENCOUNTRY CON ON CON.COUNTRYCODE=BIL.COUNTRYCODE  
LEFT JOIN GENPLACE PLACE ON PLACE.PLACECODE=BIL.CITYCODE  
LEFT JOIN GENCONSTANT BANK ON BANK.CONSTANTVALUE=BIL.BANKCODE AND CATEGORY='SP_BANKS' AND Upper(LANGUAGECODE)=Upper('EN-US')
LEFT JOIN GENPROVINCE STATE ON STATE.PROVINCECODE=BIL.STATECODE  
LEFT JOIN GENCURRENCY GEC ON GEC.CURRENCYCODE=BIL.CURRENCYCODE 

Broker :(No need Agent Name)

SELECT IB.*,LOC.HRLOCATIONNAME LOCATIONNAME,GEN.CONSTANTNAME  AS BROKERSTATUSNAME,GENC.COUNTRYNAME,PLACE.PLACENAME CITYNAME,
GPC.PROVINCENAME AS STATENAME,GENCO.CONSTANTNAME AS CURRENCYNAME,ORG.ORGANIZATION_NAME BRANCHNAME FROM IM_BROKERS IB  
LEFT JOIN GENCONSTANT  GEN ON GEN.CONSTANTVALUE=IB.BROKER_STATUS AND GEN.CATEGORY='NGIBROKERSTATUS' AND 
Upper(GEN.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN HR_LOCATIONS_D LOC ON LOC.HRLOCATION_CODE=IB.LOCATION_ID 
LEFT JOIN GENCOUNTRY GENC ON GENC.COUNTRYCODE=IB.BROKER_COUNTRY_ID
LEFT JOIN GENPLACE PLACE ON PLACE.PLACECODE=IB.BROKER_CITY_ID
LEFT JOIN GENPROVINCE GPC ON GPC.PROVINCECODE=IB.BROKER_STATE_ID 
LEFT JOIN HR_ORGANIZATIONS_D ORG ON ORG.ORGANIZATIONS_CODE=IB.BRANCHCODE  
LEFT JOIN GENCONSTANT GENCO ON GENCO.CONSTANTVALUE=IB.CURRENCYCODE  AND GENCO.CATEGORY='BILLINGCURRENCY' 
AND Upper(GENCO.LANGUAGECODE)=Upper('EN-US') 


Broker Billing Detail :

SELECT BIL.*,BRO.BROKER_NAME_EN BROKERNAME,BRO.BROKER_NAME_AR BROKERALTERNATIVENAME,
GEN.CONSTANTNAME AS AUTHORIZEDSTATUSNAME,  CON.COUNTRYNAME COUNTRYNAME,PLACE.PLACENAME CITYNAME,STATE.PROVINCENAME STATENAME,
BANK.CONSTANTNAME BANKNAME,GEC.CURRENCYNAME CURRENCYNAME FROM IM_BROKER_BILLING_DETAIL BIL   
LEFT JOIN IM_BROKERS BRO ON BRO.BROKERCODE=BIL.BROKERCODE  
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' 
AND UPPER(LANGUAGECODE)=UPPER('EN-US'))GEN ON GEN.CONSTANTVALUE=BIL.AUTHORIZEDSTATUS   
LEFT JOIN GENCOUNTRY CON ON CON.COUNTRYCODE=BIL.COUNTRYCODE  
LEFT JOIN GENPLACE PLACE ON PLACE.PLACECODE=BIL.CITYCODE 
LEFT JOIN GENCONSTANT BANK ON BANK.CONSTANTVALUE=BIL.BANKCODE AND CATEGORY='SP_BANKS' AND Upper(LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENPROVINCE STATE ON STATE.PROVINCECODE=BIL.STATECODE 
LEFT JOIN GENCURRENCY GEC ON GEC.CURRENCYCODE=BIL.CURRENCYCODE
