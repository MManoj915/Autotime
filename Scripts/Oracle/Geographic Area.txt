Geographic Area Master :

SELECT GA.*,GEN1.CONSTANTNAME STATUSNAME,'' LOADCOUNTRYSTATECITYBUTTON FROM IM_GEOGRAPHIC_AREAS GA 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=GA.STATUS  AND GEN1.CATEGORY='PROVIDERSTATUS' AND Upper(GEN1.LANGUAGECODE)=UPPER('EN-US') 
ORDER BY GA.GACODE DESC;

Country Detail :

SELECT GACON.*,CON.COUNTRYNAME FROM IM_GA_COUNTRY GACON  
LEFT JOIN GENCOUNTRY CON ON CON.COUNTRYCODE=GACON.COUNTRYCODE WHERE GACON.GACODE=100000000000000016;

Emirate Sub Detail : 

SELECT GAEMI.*,EMIRAT.PROVINCENAME EMIRATENAME FROM IM_GA_EMIRATES GAEMI  
LEFT JOIN GENPROVINCE EMIRAT ON EMIRAT.PROVINCECODE=GAEMI.EMIRATECODE;


City Sub Detail :

SELECT GACITY.*,CITY.PLACENAME CITYNAME FROM IM_GA_CITY GACITY   
LEFT JOIN GENPLACE CITY ON CITY.PLACECODE=GACITY.CITYCODE;
