Override Per Provider Type : 

SELECT POLORPP.*,GEN.CONSTANTNAME APPLICABLEFORNAME,GEN1.CONSTANTNAME PROVIDERTYPENAME,GEN2.CONSTANTNAME DEDECTABLETYPENAME,  
GEN3.CONSTANTNAME COINTYPENAME ,GEN4.CONSTANTNAME LIMITTYPENAME,GEN5.CONSTANTNAME OONDEDECTABLETYPENAME,
GEN6.CONSTANTNAME OONCOINTYPENAME FROM IM_POLOVERRIDEPERPROVIDERDTL POLORPP   
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=POLORPP.APPLICABLEFOR AND GEN.CATEGORY='APPLICABLE' AND Upper(GEN.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=POLORPP.PROVIDERTYPE  AND GEN1.CATEGORY='PROVIDERTYPE' AND Upper(GEN1.LANGUAGECODE)=Upper('EN-US')
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=POLORPP.DEDECTABLETYPE  AND GEN2.CATEGORY='VALUETYPE' AND Upper(GEN2.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN4 ON GEN4.CONSTANTVALUE=POLORPP.LIMITTYPE  AND GEN4.CATEGORY='VALUETYPE' AND Upper(GEN4.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN3 ON GEN3.CONSTANTVALUE=POLORPP.CO_IN_TYPE AND GEN3.CATEGORY='VALUETYPE' AND Upper(GEN3.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN5 ON GEN5.CONSTANTVALUE=POLORPP.OONDEDECTABLETYPE AND GEN5.CATEGORY='VALUETYPE' AND Upper(GEN5.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN6 ON GEN6.CONSTANTVALUE=POLORPP.OONCO_IN_TYPE AND GEN6.CATEGORY='VALUETYPE' AND Upper(GEN6.LANGUAGECODE)=Upper('EN-US');

Over ride per provider :

SELECT POLORP.*,GEN.CONSTANTNAME APPLICABLEFORNAME,GEN2.CONSTANTNAME DEDECTABLETYPENAME,PRO.PROVIDERNAME,  
GEN3.CONSTANTNAME COINTYPENAME ,GEN4.CONSTANTNAME LIMITTYPENAME,GEN5.CONSTANTNAME OONDEDECTABLETYPENAME ,
GEN6.CONSTANTNAME OONCOINTYPENAME FROM IM_POLCYOVERRIDEPROVIDERDET POLORP   
LEFT JOIN IM_PROVIDERS PRO ON PRO.PROVIDERCODE=POLORP.PROVIDERCODE  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=POLORP.APPLICABLEFOR AND GEN.CATEGORY='APPLICABLE' AND Upper(GEN.LANGUAGECODE)=Upper(Upper('EN-US')) 
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=POLORP.DEDECTABLETYPE  AND GEN2.CATEGORY='VALUETYPE' AND Upper(GEN2.LANGUAGECODE)=Upper(Upper('EN-US')) 
LEFT JOIN GENCONSTANT GEN4 ON GEN4.CONSTANTVALUE=POLORP.LIMITTYPE  AND GEN4.CATEGORY='VALUETYPE' AND Upper(GEN4.LANGUAGECODE)=Upper(Upper('EN-US')) 
LEFT JOIN GENCONSTANT GEN3 ON GEN3.CONSTANTVALUE=POLORP.CO_IN_TYPE AND GEN3.CATEGORY='VALUETYPE' AND Upper(GEN3.LANGUAGECODE)=Upper(Upper('EN-US')) 
LEFT JOIN GENCONSTANT GEN5 ON GEN5.CONSTANTVALUE=POLORP.OONDEDECTABLETYPE AND GEN5.CATEGORY='VALUETYPE' AND Upper(GEN5.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN6 ON GEN6.CONSTANTVALUE=POLORP.OONCO_IN_TYPE AND GEN6.CATEGORY='VALUETYPE' AND Upper(GEN6.LANGUAGECODE)=Upper('EN-US');
