SELECT BIL.*,MEM.MEMBER_ID MEMBERID ,(MEM.FIRST_NAME||' '||MEM.LAST_NAME)  MEMBER_NAME, MEM.ARABIC_FIRST_NAME ALTERNATENAME,
GEN.CONSTANTNAME AS AUTHORIZEDSTATUSNAME, CON.COUNTRYNAME COUNTRYNAME,PLACE.PLACENAME CITYNAME,STATE.PROVINCENAME STATENAME,
BANK.CONSTANTNAME BANKNAME,GEC.CURRENCYNAME CURRENCYNAME FROM IM_MEMBER_BILLING_DETAIL BIL 
LEFT JOIN IM_MEMBERS MEM ON MEM.MEMBER_CODE=BIL.MEMBER_CODE  
LEFT JOIN GENCONSTANT BANK ON BANK.CONSTANTVALUE=BIL.BANKCODE AND CATEGORY='SP_BANKS' AND Upper(LANGUAGECODE)=UPPER('EN-US')  
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' AND  Upper(LANGUAGECODE)=UPPER('EN-US'))GEN ON GEN.CONSTANTVALUE=BIL.AUTHORIZEDSTATUS 
LEFT JOIN GENCOUNTRY CON ON CON.COUNTRYCODE=BIL.COUNTRYCODE  
LEFT JOIN GENPLACE PLACE ON PLACE.PLACECODE=BIL.CITYCODE  
LEFT JOIN GENPROVINCE STATE ON STATE.PROVINCECODE=BIL.STATECODE   
LEFT JOIN GENCURRENCY GEC ON GEC.CURRENCYCODE=BIL.CURRENCYCODE
