SELECT  'NATIONAL GENERAL INSURANCE CO. - NGI' PAYERS,SYSDATE TDATE,
NVL(POL.POLICYNAME,IPOL.POLICYNAME) POLICYNAME,NVL(POL.POLICYID,IPOL.POLICYID) POLICYID,
'ENBD-TANFEETH DXB' SUBCONTRACTNAME,'27241' STAFFID,
INITCAP(IM.FIRST_NAME) PRINCIPAL,IMC.CARDNO,TO_CHAR(CPH.CPNO) TRANSACTIONNO,'188888640' PO,
IMB.FROMID BATCH,IMB.RECEIVEDON RECEPTIONDATE,'NGI-RB5552-05NOV14-40-INT' LABEL,
'BSC(500K-D:35@CLNCS@HSP-DT:20%-MAT-MD-GN' PRODUCT,IM.EMIRATES_ID IDCARDNO,ROWNUM REFERENCENO,
ACT.ACTIVITYSTARTDATE DETAILDATE,NVL(CPH.OUTOFNETWORKPROVIDERNAME,PRO.PROVIDERNAME) PROVIDERNAME,ACT.REQUESTAMOUNT CLAIMED,
ACT.DENIAL_VALUE,(ACT.TOTAL + ACT.DEDUCTIBLE_AMOUNT + ACT.COINSURANCE_AMOUNT) APPROVED,ACT.DEDUCTIBLE_AMOUNT DEDUCTIBLE,
ACT.COINSURANCE_AMOUNT COINSURANCE,ACT.TOTAL PAYERSHARE,GEN.SHORTNAME CURR,
ACT.ACTIVITY_DETAIL_CODE,CPH.ACR_FORM_NO,'' USERNAME,
SPELLNUM((SELECT NVL(SUM(NEWACT.TOTAL),0) FROM IM_CLAIM_PROCESS_DETAIL NEWACT WHERE NEWACT.CLAIM_CODE = ACT.CLAIM_CODE)) AMOUNTINWORDS,
ACT.REIMBURSEMENTAMT,ACT.RATE,CPH.CLAIM_CODE,CPH.ENCOUNTER_START_DATE TREATMENTDATE,
(SELECT LISTAGG (IMVD.SHORTDESC, ',') WITHIN GROUP (ORDER BY IMVD.SHORTDESC) FROM IM_CLAIM_PROCESS_DETAIL DTL
LEFT JOIN IM_ACTIVITY_TYPES  ACTTYP ON ACTTYP.TYPECODE=DTL.ACTIVITY_CODE
JOIN IM_VERSION_DETALIS IMVD ON IMVD.DETAILCODE = DTL.VERSIONDETAILCODE
WHERE ACTTYP.VALUE = -1 AND DTL.CLAIM_CODE = CPH.CLAIM_CODE GROUP BY DTL.CLAIM_CODE ) ICDCODES,
IAR.DESCRIPTION,'' DECLINEDREASON,IMVD.SHORTDESC BENEFIT_NAME,1 PAGEHEADER
FROM IM_CLAIM_PROCESS_HEADER CPH
JOIN IM_CLAIM_PROCESS_DETAIL ACT ON ACT.CLAIM_CODE = CPH.CLAIM_CODE AND ACT.TYPE = 0
JOIN IM_VERSION_DETALIS IMVD ON IMVD.DETAILCODE =  ACT.VERSIONDETAILCODE
LEFT JOIN IM_ACTIVITY_REMARKS_VW IAR ON IAR.ACTIVITY_DETAIL_CODE = ACT.ACTIVITY_DETAIL_CODE
--JOIN ADMUSER ADM ON ADM.USERCODE = AUSERCODE
LEFT JOIN (SELECT ENDORESMENTCODE,CLAIMCODE FROM IM_ENDPOSTCLAIMDTL WHERE ENDORESMENTCODE IN
(SELECT ENDORESMENTCODE FROM IM_ENDORSEMENTPOSTING WHERE AUTHORIZEDSTATUS = 1) AND NVL(INCLUDE_CLAIM,0) = 0) IMEPD ON IMEPD.CLAIMCODE = CPH.CLAIM_CODE
LEFT JOIN IM_ENDORSEMENTPOSTING IMEP ON IMEP.ENDORESMENTCODE = IMEPD.ENDORESMENTCODE AND IMEP.AUTHORIZEDSTATUS = 1
JOIN IM_MEMBERS IM ON IM.MEMBER_CODE = CPH.MEMBER_CODE
LEFT JOIN IM_PROVIDERS PRO ON PRO.PROVIDERCODE = CPH.PROVIDER_CODE
LEFT JOIN IM_CORDPRINT IMC ON IMC.MEMBERCODE = CPH.MEMBER_CODE  AND IMC.POLICYCODE = CPH.POLICYCODE  AND CPH.CATEGORY_CODE = IMC.CATEGORYCODE
LEFT JOIN GENCURRENCY GEN ON GEN.CURRENCYCODE = ACT.CURRENCY
JOIN IM_MEMBERPOLICY IMP ON IMP.MEMBERCODE = CPH.MEMBER_CODE AND CPH.POLICYCODE = IMP.POLICYCODE  AND CPH.CATEGORY_CODE = IMP.CATEGORYCODE
LEFT JOIN IM_POLICY POL ON POL.POLICYCODE = IMP.POLICYCODE AND IMP.TYPEE = 1
LEFT JOIN IM_INDIVIDUALPOLICY IPOL ON IPOL.INDIVIDUALPOLICYCODE = IMP.POLICYCODE AND IMP.TYPEE = 2
JOIN IM_CLAIMBATCH IMB ON IMB.CLAIMBATCHCODE = CPH.BATCHCODE
WHERE CPH.CLAIM_CODE = APRIKEY AND CPH.SUBMITTORA = 0 AND CPH.REQUEST_TYPE = 3
ORDER BY CPH.CLAIM_CODE
