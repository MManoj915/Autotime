SELECT CA.*,PRO.PROVIDERID,PRO.PROVIDERNAME,MT.CATEGORIES_TEMPLATE  CATEGORIES_TEMPLATE_NAME,
GEN.CONSTANTNAME AS AUTHORIZEDSTATUSNAME 
FROM IM_MANDATORYTARIFF_ASSIGN CA  
JOIN ADMCOMPANY ADMCOM ON ADMCOM.COMPANYCODE=CA.COMPANYCODE AND CA.COMPANYCODE =1  
LEFT JOIN IM_PROVIDERS  PRO ON PRO.PROVIDERCODE=CA.PROVIDERCODE  
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' AND LANGUAGECODE='EN-US')GEN 
ON GEN.CONSTANTVALUE=CA.AUTHORIZEDSTATUS   
LEFT JOIN IM_MANDATORY_TARIFF MT ON MT.MANDATORY_TARIFF_CODE =CA.CATEGORIES_TEMPLATE_CODE  ORDER BY CA.ASSIGN_CODE DESC

SELECT CAD.*, GEN.CONSTANTNAME TREATMENTTYPENAME,GEN2.CONSTANTNAME FACTORTYPENAME, GEN1.CONSTANTNAME CALMETHODNAME 
FROM IM_CODE_CATASSIGN_DETAIL CAD  
LEFT JOIN GENCONSTANT GEN ON  GEN.CATEGORY= 'APPLICABLE' AND GEN.LANGUAGECODE='EN-US' AND 
GEN.CONSTANTVALUE=CAD.TREATMENTTYPECODE  
LEFT JOIN GENCONSTANT GEN1 ON  GEN1.CATEGORY= 'SLABCALCULATIONMODE' AND GEN1.LANGUAGECODE='EN-US' AND 
GEN1.CONSTANTVALUE=CAD.CALMETHODCODE   
LEFT JOIN GENCONSTANT GEN2 ON  GEN2.CATEGORY= 'LBLFACTORBASE' AND GEN2.LANGUAGECODE='EN-US' AND 
GEN2.CONSTANTVALUE=CAD.FACTORTYPECODE
