SELECT IME.*,MEM.MEMBER_ID MEMBERIDNAME,(MEM.FIRST_NAME||' '||MEM.LAST_NAME)  MEMBER_NAME,MEM.ARABIC_FIRST_NAME ALTERNATENAME,
ACT.TYPENAME,VER.VERSIONNAME,VERDTL.SHORTDESC DETAILNAME,GEN1.CONSTANTNAME AS EXCLUSIONTYPENAME,
GEN2.CONSTANTNAME AS SOURCENAME FROM IM_MEMBERS_EXCLUSIONS IME 
LEFT JOIN IM_MEMBERS MEM ON MEM.MEMBER_CODE=IME.MEMBER_CODE                          
LEFT JOIN IM_ACTIVITY_TYPES  ACT ON ACT.TYPECODE=IME.TYPECODE 
LEFT JOIN IM_TYPE_VERSIONS  VER ON VER.VERSIONCODE=IME.VERSIONCODE 
LEFT JOIN IM_VERSION_DETALIS  VERDTL ON VERDTL.DETAILCODE=IME.DETAILCODE 
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=IME.EXCLUSTION_TYPE AND GEN1.CATEGORY='NGIADDORREMOVE' AND 
Upper(GEN1.LANGUAGECODE)=UPPER('EN-US') 
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=IME.EXCLUSTION_SOURCE  AND GEN2.CATEGORY='NGISOURCE' AND 
GEN2.LANGUAGECODE=UPPER('EN-US')
