Policy Category Detail :

SELECT PCD.*,GC.CATEGORY_NAME CATEGORYNAME,PLA.PLAN_NAME PLANNAME,II.INS_NAME FRONTINGSETTINGNAME FROM IM_POLICY_CATEGORYDTL PCD 
LEFT JOIN IM_CATEGORIES GC ON GC.CATEGORY_CODE=PCD.CATEGORYCODE 
LEFT JOIN IM_PLANS PLA ON PLA.PLAN_CODE=PCD.PLANCODE 
LEFT JOIN IM_INSURER II ON II.INSCODE=PCD.FRONTINGSETTING;


Policy Member Detail : 

SELECT IQM.*,IME.MEMBER_ID  MEMBERIDNAME,GEN.CONSTANTNAME AS GENDERNAME,GEN1.CONSTANTNAME MEMBERRELATIONNAME,
GEN2.CONSTANTNAME MARITALSTATUSNAME,GEN5.CONSTANTNAME SALARYBANDNAME,  GEN3.CONSTANTNAME AS NATIONALITYNAME,
FLOOR(MONTHS_BETWEEN(SYSDATE,IQM.DATE_OF_BIRTH)/12) AS AGE,GC.CATEGORY_NAME CATEGORYNAME ,
(SELECT COUNT(*) FROM IM_POLICY_PREMIUM DET WHERE DET.POLICY_MEM_CODE=IQM.POLICY_MEM_CODE AND DET.PREMIUM_TYPES=0) 
AS ISBASEPREMIUMAVAIL,PARGRO.GROUP_NAME AS PARENTGROUPNAME,  
(SELECT SUM(PREMIUM_VALUE)  FROM IM_POLICY_PREMIUM WHERE POLICY_MEM_CODE=IQM.POLICY_MEM_CODE ) PREMIUMS FROM 
IM_POLICY_MEMBERS IQM  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=IQM.MEMBER_GENDER  AND GEN.CATEGORY='GENDER' AND Upper(LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=IQM.MEMBER_RELATION AND GEN1.CATEGORY='MEMBERRELATION' AND Upper(GEN1.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=IQM.MARITAL_STATUS AND GEN2.CATEGORY='MARITAL_STATUS' AND Upper(GEN2.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN3 ON GEN3.CONSTANTVALUE=IQM.NATIONALITY AND GEN3.CATEGORY='FND_NATIONALITY' AND Upper(GEN3.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN GENCONSTANT GEN5 ON GEN5.CONSTANTVALUE=IQM.SALARYBAND AND GEN5.CATEGORY='SALARYBANDTYPE' AND Upper(GEN5.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN IM_MEMBERS IME ON IME.MEMBER_CODE= IQM.MEMBERCODE   LEFT JOIN IM_CATEGORIES GC ON GC.CATEGORY_CODE=IQM.MEMBER_CATEGORY   
LEFT JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='GENDER' AND Upper(LANGUAGECODE)=Upper('EN-US'))GEN4 ON GEN4.CONSTANTVALUE=IQM.MEMBER_GENDER  
LEFT JOIN IM_GROUPS PARGRO ON PARGRO.GROUP_CODE=IQM.PARENTGROUPCODE 
ORDER BY NVL(IQM.PARENTID,IQM.MEMBERID),NVL2(IQM.PARENTID,0,1) DESC;

Premium Sub detail :

SELECT PCO.*,GEN.CONSTANTNAME PREMIUMTYPENAME,GEN1.CONSTANTNAME CALCMETHODNAME,IBC.BENEFIT_ID BENEFITID, 
IBC.BENEFIT_NAME BENEFITNAME FROM IM_POLICY_PREMIUM PCO  
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=PCO.PREMIUM_TYPES AND GEN.CATEGORY='PREMIUMTYPE' AND Upper(GEN.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=PCO.CALC_METHOD AND GEN1.CATEGORY='FINECALCULATIONTYPE' AND Upper(GEN1.LANGUAGECODE)=Upper('EN-US') 
LEFT JOIN IM_BENEFIT_CODES IBC ON IBC.BENEFIT_CODE=PCO.BENEFITCODE;

Policy Commission Detail :

SELECT PCO.*,GEN.CONSTANTNAME COMISSIONTYPENAME,GEN1.CONSTANTNAME CALCMETHODNAME,COM.COMMISIONNAME,
COMBILL.COMMISSIONNAME COMMISSIONNAMEE FROM IM_POLICY_COMMISSION PCO   
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=PCO.COMISSION_TYPES AND GEN.CATEGORY='NGICOMMISSIONTYPE' AND Upper(GEN.LANGUAGECODE)=Upper('EN-US')  
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=PCO.CALC_METHOD AND GEN1.CATEGORY='FINECALCULATIONTYPE' AND Upper(GEN1.LANGUAGECODE)=UppEr('EN-US')  
LEFT JOIN IM_GENERAL_COMM_BILLING_DETAIL COMBILL ON COMBILL.GENERALCOMMBILLINGCODE=PCO.GENERALCOMMBILLINGCODE 
LEFT JOIN (SELECT AGE.AGENTCODE COMMISIONTOCODE, AGE.AGENT_ID COMMISIONID, AGE.AGENT_NAME_EN COMMISIONNAME ,1 TYPECODE FROM 
IM_AGENTS AGE UNION ALL SELECT BRO.BROKERCODE COMMISIONTOCODE, BRO.BROKER_ID COMMISIONID, BRO.BROKER_NAME_EN COMMISIONNAME,
0 TYPECODE FROM IM_BROKERS BRO) COM ON  COM.COMMISIONTOCODE=PCO.COMMISSIONTO  AND COM.TYPECODE=PCO.COMISSION_TYPES;

Exclusion Detail : 

SELECT EX.*,EXGRO.EXC_GROUP_NAME EXCLUTIONGROUPNAME FROM IM_POLICY_BS_EXCLUSION EX   
LEFT JOIN IM_EXCLUSIONS_GROUPS_MASTER  EXGRO ON EXGRO.EXC_GRP_CODE=EX.EXC_GROUP_CODE;


Exclusion Standard Detail :

SELECT IBES.*,VER.CODE, VER.SHORTDESC SHORTDESCRIPTION, VER.LONGDESC LONGDESCRIPTION  FROM IM_POLICY_EXCLUSION_STANDCODE  IBES  
LEFT JOIN IM_VERSION_DETALIS VER ON VER.DETAILCODE =IBES.STANDARD_CODE;
