SELECT IMP.*,MEM.MEMBER_ID MEMBERID,(MEM.FIRST_NAME||' '||MEM.LAST_NAME) MEMBERNAME,PO.POLICYID,POL.POLICYNAME,
POL.STARTDATE,POL.ENDDATE,POL.BASE_PERIMUM, GENNP.CONSTANTNAME PERIMUMNAME,GENP.CONSTANTNAME ANNUALLIMITTYPENAME,
POL.ALTLIMIT,GEN1P.CONSTANTNAME INPATIENTNAME,POL.IPLIMIT,GEN2P.CONSTANTNAME OUTPATIENTNAME,POL.OPLIMIT , 
GROP.GROUP_NAME GROUPNAME,GEN.CONSTANTNAME AS AUTHORIZEDSTATUSNAME,' 'VIEWMEMBERBUTTON,' 'VIEWPOLICYMEMBER,
GEE.CONSTANTNAME POLICYTYPENAME, IPOL.POLICYNAME INDIVIDUALPOLICYNAME, IPOL.POLICYID IPPOLICYID,   
POL.GROUPCODE,INS.INS_NAME INSURANCENAME,GEN8.CONSTANTNAME CONTRACTTYPENAME
FROM IM_ADDMEMBERPOLICY IMP   
JOIN ADMCOMPANY ADMCOM ON ADMCOM.COMPANYCODE=IMP.COMPANYCODE AND ADMCOM.COMPANYCODE= 1 
LEFT JOIN IM_MEMBERS MEM ON MEM.MEMBER_CODE=IMP.MEMBERCODE   
JOIN (SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE CATEGORY='AUTHORIZEDSTATUS' AND Upper(LANGUAGECODE)='EN-US')GEN 
ON GEN.CONSTANTVALUE=IMP.AUTHORIZEDSTATUS 
LEFT JOIN (SELECT PO.POLICYCODE POLICYCODE, PO.POLICYID,PO.POLICYNAME,PO.STARTDATE,PO.ENDDATE,PO.BASE_PERIMUM,
PO.ALTLIMIT ALTLIMIT,PO.IPLIMIT IPLIMIT,PO.OPLIMIT OPLIMIT,1 TYPECODE,PO.CATEGORYCODE,PO.GROUPCODE,
PO.PERIMUMCODE,PO.ANNUALLIMITTYPE ANNUALLIMITTYPE,PO.INPATIENT INPATIENT,PO.OUTPATIENT OUTPATIENT,
PO.INSURANCECO ,PO.CONTRACTTYPE FROM IM_POLICY PO UNION ALL 
SELECT IPO.INDIVIDUALPOLICYCODE POLICYCODE,IPO.POLICYID, IPO.POLICYNAME,IPO.STARTDATE,IPO.ENDDATE,IPO.BASE_PERIMUM,
IPO.ALTLIMITS ALTLIMIT,IPO.IPTLIMITS IPLIMIT,IPO.OPTLIMITS,2 TYPECODE,IPO.CATEGORYCODE,IPO.GROUPCODE,IPO.PERIMUMCODE,
IPO.ANUALLIMITSTYPECODE ANNUALLIMITTYPE,IPO.INPATENTSLIMITSTYPECODE INPATIENT,IPO.OUTPATENTSLIMITSTYPECODE OUTPATIENT,
IPO.INSURANCECO,IPO.CONTRACTTYPE FROM IM_INDIVIDUALPOLICY IPO)POL ON 
POL.POLICYCODE=DECODE(IMP.POLICYTYPE,1,IMP.POLICYCODE,2,IMP.INDIVIDUALPOLICYCODE)AND POL.TYPECODE=IMP.POLICYTYPE    
LEFT JOIN IM_CATEGORIES CATP ON CATP.CATEGORY_CODE=POL.CATEGORYCODE    
LEFT JOIN IM_GROUPS GROP ON GROP.GROUP_CODE =POL.GROUPCODE   
LEFT JOIN GENCONSTANT  GENNP ON GENNP.CONSTANTVALUE=POL.PERIMUMCODE AND GENNP.CATEGORY='PERIMUMS' AND 
Upper(GENNP.LANGUAGECODE)='EN-US'  
LEFT JOIN GENCONSTANT GENP ON GENP.CONSTANTVALUE=POL.ANNUALLIMITTYPE AND GENP.CATEGORY='CALCMETHOD' AND 
Upper(GENP.LANGUAGECODE)='EN-US'   
LEFT JOIN GENCONSTANT GEN1P ON GEN1P.CONSTANTVALUE=POL.INPATIENT AND GEN1P.CATEGORY='CALCMETHOD' AND 
Upper(GEN1P.LANGUAGECODE)='EN-US'   
LEFT JOIN GENCONSTANT GEN2P ON GEN2P.CONSTANTVALUE=POL.OUTPATIENT AND  GEN2P.CATEGORY='CALCMETHOD' AND 
Upper(GEN2P.LANGUAGECODE)='EN-US'  
LEFT JOIN GENCONSTANT GEE ON GEE.CONSTANTVALUE=IMP.POLICYTYPE AND  GEE.CATEGORY='MEMBERPOLICYTYPE' 
AND Upper(GEE.LANGUAGECODE)='EN-US' 
LEFT JOIN IM_INDIVIDUALPOLICY IPOL ON IPOL.INDIVIDUALPOLICYCODE= IMP.INDIVIDUALPOLICYCODE  
LEFT JOIN IM_POLICY PO ON PO.POLICYCODE= IMP. POLICYCODE  
LEFT JOIN (SELECT * FROM  IM_INSURER) INS ON INS.INSCODE=POL.INSURANCECO 
LEFT JOIN GENCONSTANT GEN8 ON GEN8.CONSTANTVALUE=POL.CONTRACTTYPE AND GEN8.CATEGORY='CONTRACTINGTYPE' AND 
Upper(GEN8.LANGUAGECODE)='EN-US' ORDER BY IMP.ADDMEMBERPOLICYCODE DESC;

Member Detail :

SELECT IQM.*,GEN.CONSTANTNAME AS GENDERNAME,GEN1.CONSTANTNAME MEMBERRELATIONNAME,GEN2.CONSTANTNAME MARITALSTATUSNAME,
GEN3.CONSTANTNAME AS NATIONALITYNAME,FLOOR(MONTHS_BETWEEN(SYSDATE,IQM.DATE_OF_BIRTH)/12) AS AGE,GC.CATEGORY_NAME CATEGORYNAME,
PARGRO.GROUP_NAME AS PARENTGROUPNAME,IME.CARDID FROM IM_ADDMEMBERDET IQM    
LEFT JOIN IM_MEMBERS IME ON IME.MEMBER_CODE= IQM.MEMBERCODE     
LEFT JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE=IQM.GENDERCODE  AND GEN.CATEGORY='GENDER' AND Upper(LANGUAGECODE)='EN-US'   
LEFT JOIN GENCONSTANT GEN1 ON GEN1.CONSTANTVALUE=IQM.MEMBERRELATIONCODE AND GEN1.CATEGORY='MEMBERRELATION' AND 
Upper(GEN1.LANGUAGECODE)='EN-US'   
LEFT JOIN GENCONSTANT GEN2 ON GEN2.CONSTANTVALUE=IQM.MARITALSTATUSCODE AND GEN2.CATEGORY='MARITAL_STATUS' AND 
Upper(GEN2.LANGUAGECODE)='EN-US'   
LEFT JOIN GENCONSTANT GEN3 ON GEN3.CONSTANTVALUE=IQM.NATIONALITYCODE AND GEN3.CATEGORY='FND_NATIONALITY' AND 
Upper(GEN3.LANGUAGECODE)='EN-US'  
LEFT JOIN IM_CATEGORIES GC ON GC.CATEGORY_CODE=IQM.CATEGORYCODE 
LEFT JOIN IM_GROUPS PARGRO ON PARGRO.GROUP_CODE=IQM.PARENTGROUPCODE
ORDER BY NVL(IQM.PARENTID,IQM.MEMBERID),NVL2(IQM.PARENTID,0,1) DESC;
