SOURCE :

SELECT CONSTANTVALUE,CONSTANTNAME FROM GENCONSTANT WHERE LANGUAGECODE='en-US' AND 
CATEGORY='PRODUCTIONSOURCE'

BATCH NO:

SELECT * FROM IM_PRODUCTIONBATCH_VW


POLICYCODE:

SELECT DISTINCT POLICYID,POLICYNAME,POLICYCODE,BATCHNO,SOURCE FROM IM_TPAPOLICIESBULK_VW POL   

DETAIL:

SELECT VW.*,ROWNUM AS PKEY,0 Selec FROM(SELECT DISTINCT VW.POLICYID,VW.POLICYNAME,VW.POLICYCODE,SOURCE,
VW.COMPANYCODE,BATCHNO,GEN.CONSTANTNAME BILLINGTYPE FROM IM_TPAPOLICIESBULK_VW VW  
JOIN IM_POLICY POL ON POL.POLICYCODE = VW.POLICYCODE  
JOIN GENCONSTANT GEN ON GEN.CONSTANTVALUE = POL.BILLINGTYPE AND 
GEN.CATEGORY = 'POLICYBILLINGTYPE' AND Upper(GEN.LANGUAGECODE) = 'EN-US')VW
