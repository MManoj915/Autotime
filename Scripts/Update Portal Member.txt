DECLARE L_ID NVARCHAR2(500);
BEGIN
FOR HDR IN(SELECT * FROM IM_INDIVIDUALPOLICY WHERE 
OLDPOLICYCODE IN
(SELECT INDIVIDUALPOLICYCODE FROM IM_INDIVIDUALPOLICY WHERE PORTALMEMBERID IS NOT NULL)
AND PORTALMEMBERID IS NULL) LOOP

SELECT PORTALMEMBERID INTO L_ID FROM IM_INDIVIDUALPOLICY WHERE INDIVIDUALPOLICYCODE = HDR.OLDPOLICYCODE;
UPDATE IM_INDIVIDUALPOLICY SET PORTALMEMBERID = L_ID WHERE INDIVIDUALPOLICYCODE = HDR.INDIVIDUALPOLICYCODE
AND PORTALMEMBERID IS NULL;
COMMIT;


END LOOP;
END;
