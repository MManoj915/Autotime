SELECT EC.SYS_CLAIM_ID,HDR.ACR_FORM_NO,IMP.APOLICYID,IMP.APOLICYNAME,IMP.ACARDID,
EC.CLAIM_ID,HDR.REQUEST_AMOUNT,ECB.FILE_NAME,EC.TRANSACTION_DATE,
Decode(Nvl(EC.RESUBMISSION_COMMENT,'0'),'0','Submission','ReSubmission') CLAIMTYPE,
HDR.PAYMENTREFNO DOCUMENTNO,HDR.PAYMENTREFDATE DOCUMENTDATE,HDR.APPROVED_AMOUNT,HDR.SUBMITTORA,
HDR.RAUPLOADDATE,EC.AUTOREJECT,Decode(EC.SYS_CLAIM_STATUS,-1,'Claim Interface','') CLAIMSTATUS,
PRO.PROVIDERCODE
FROM ECLAIMS EC
LEFT JOIN ECLAIM_BATCH_FILES ECB ON ECB.SYS_ID = EC.SYS_FILE_ID
LEFT JOIN IM_CLAIM_PROCESS_HEADER HDR ON HDR.MAP_VALUE = EC.SYS_CLAIM_ID
LEFT JOIN IM_MEMBERPOLICY IMP ON UPPER(TRIM(REPLACE(IMP.ACARDID,' ')))=UPPER(TRIM(REPLACE(EC.MEMBER_ID,' ','')))
LEFT JOIN IM_PROVIDER_LICENCES IPL ON UPPER(TRIM(REPLACE(IPL.OTHERLICNECEID,' ')))  = UPPER(TRIM(REPLACE(EC.SENDER_ID,' ','')))
LEFT JOIN IM_PROVIDERS PRO ON PRO.PROVIDERCODE = IPL.PROVIDERCODE
WHERE DISPOSITION_FLAG <> '0' AND HDR.REQUEST_TYPE = 1
