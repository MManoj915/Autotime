SELECT PRO.PROVIDERID,PRO.PROVIDERNAME,Sum(CPT9),SUM(CPT10),SUM(CPT11),SUM(OTHERCPT),Sum(LABS),Sum(RADIOLOGY),Sum(DRUGS)
FROM
(SELECT PRO.PROVIDERID,PRO.PROVIDERNAME,
Sum(REQUESTAMOUNT)  CPT9,0 CPT10,0 CPT11,0 OTHERCPT,0 Labs,0 Radiology,0 Drugs,
To_Char(HDR.PAYMENTREFDATE,'MM') PAYMENTMONTH FROM ECLAIMS EC
JOIN IM_CLAIM_PROCESS_HEADER HDR ON HDR.MAP_VALUE=EC.SYS_CLAIM_ID 
JOIN IM_PROVIDERS PRO ON PRO.PROVIDERCODE = HDR.PROVIDER_CODE 
JOIN IM_CLAIM_PROCESS_DETAIL DTL ON DTL.CLAIM_CODE = HDR.CLAIM_CODE
JOIN IM_BENEFIT_CODES IBM ON IBM.BENEFIT_CODE = DTL.BENEFIT_CODE
WHERE DISPOSITION_FLAG = 'PRODUCTION' AND IBM.BENEFIT_CODE=10000000000000042 AND 
To_Date(ENCOUNTER_START,'DD/MM/RRRR') BETWEEN To_Date('01/01/2017','DD/MM/RRRR')
AND  To_Date('31/12/2017','DD/MM/RRRR')  AND HDR.MAP_VALUE=EC.SYS_CLAIM_ID
AND DTL.VERSIONDETAILCODE IN(SELECT DETAILCODE FROM IM_VERSION_DETALIS WHERE CODE='9')
GROUP BY PRO.PROVIDERCODE,PRO.PROVIDERID,PRO.PROVIDERNAME,To_Char(HDR.PAYMENTREFDATE,'MM')
UNION ALL
