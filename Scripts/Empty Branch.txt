BEGIN
FOR HDR IN
(SELECT INDIVIDUALPOLICYCODE,OWNERCODE,POLICYID,INDIPOLICYSOURCECODE,NVL(P.BRANCHCODE,100000000000000003) BRANCHCODE,NVL(SHORT_NAME,'HDO') BRANCHNAME,
AGN.AGENTCODE FROM IM_INDIVIDUALPOLICY P
LEFT JOIN HR_ORGANIZATIONS_D H ON H.ORGANIZATIONS_CODE = P.BRANCHCODE 
LEFT JOIN IM_AGENTS AGN ON AGN.BRANCHCODE = P.BRANCHCODE
WHERE STARTDATE BETWEEN TO_DATE('01/01/2017','DD/MM/RRRR')
AND TO_DATE(SYSDATE,'DD/MM/RRRR') AND P.BRANCHCODE IS NULL
AND AUTHORIZEDSTATUS = 1  ) LOOP
UPDATE IM_INDIVIDUALPOLICY SET BRANCHCODE = HDR.BRANCHCODE WHERE INDIVIDUALPOLICYCODE = HDR.INDIVIDUALPOLICYCODE;
COMMIT;
END LOOP;
END;
