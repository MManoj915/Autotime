

BEGIN 
FOR HDR IN(SELECT O.* FROM IM_INDIVIDUALPOLICY O
JOIN IM_INDIVIDUALPOLICY IPOL ON IPOL.OLDPOLICYCODE = O.INDIVIDUALPOLICYCODE
WHERE O.PORTALMEMBERID IS NOT NULL) LOOP
UPDATE IM_INDIVIDUALPOLICY SET PORTALMEMBERID = HDR.PORTALMEMBERID
WHERE OLDPOLICYCODE = HDR.INDIVIDUALPOLICYCODE;
COMMIT;
END LOOP;
END;
