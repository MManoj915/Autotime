SELECT J.*,G.CONSTANTNAME MANAGEDBY FROM IM_JV_MANAGEDBY J 
LEFT JOIN IM_NGI_JVPOSTING JV ON JV.REF_NO = J.DOCNO
LEFT JOIN IM_POLICY PO ON PO.POLICYCODE = JV.POLICYCODE 
LEFT JOIN GENCONSTANT G ON G.CATEGORY = 'NGIQUOTATIONTYPE' AND G.CONSTANTVALUE = PO.OWNERCODE AND G.LANGUAGECODE = 'en-US'
WHERE JV.POLICYTYPE = 0 AND JV.REF_CODE IS NOT NULL
UNION ALL
SELECT J.*,G.CONSTANTNAME MANAGEDBY FROM IM_JV_MANAGEDBY J 
LEFT JOIN IM_NGI_JVPOSTING JV ON JV.REF_NO = J.DOCNO
LEFT JOIN IM_INDIVIDUALPOLICY PO ON PO.INDIVIDUALPOLICYCODE = JV.POLICYCODE 
LEFT JOIN GENCONSTANT G ON G.CATEGORY = 'NGIQUOTATIONTYPE' AND G.CONSTANTVALUE = PO.OWNERCODE AND G.LANGUAGECODE = 'en-US'
WHERE JV.POLICYTYPE = 1 AND JV.REF_CODE IS NOT NULL
UNION ALL
SELECT J.*,G.CONSTANTNAME MANAGEDBY FROM IM_JV_MANAGEDBY J 
LEFT JOIN IM_NGI_JVPOSTING JV ON JV.REF_NO = J.DOCNO
LEFT JOIN IM_INDIVIDUALPOLICY PO ON PO.INDIVIDUALPOLICYCODE = JV.POLICYCODE 
LEFT JOIN GENCONSTANT G ON G.CATEGORY = 'NGIQUOTATIONTYPE' AND G.CONSTANTVALUE = PO.OWNERCODE AND G.LANGUAGECODE = 'en-US'
WHERE  JV.REF_CODE IS   NULL
